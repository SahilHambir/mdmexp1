Producer Consumer:

import threading
import random
import time

# Fixed size of buffer
BUFFER_SIZE = 5

# Shared buffer initialized with 0
buffer = [0] * BUFFER_SIZE

# Index positions and count
# in_index  -> where producer inserts item
# out_index -> where consumer removes item
# count     -> number of items currently in buffer
in_index = 0
out_index = 0
count = 0

# Mutex lock for mutual exclusion
# Only one thread can access critical section at a time
mutex = threading.Lock()

# Condition variable for buffer full condition
not_full = threading.Condition(mutex)

# Condition variable for buffer empty condition
not_empty = threading.Condition(mutex)


# Producer Function
def producer():
    global in_index, out_index, count

    while True:

        # Produce a random item between 0 to 99
        item = random.randint(0, 99)

        # Enter critical section
        with not_full:

            # If buffer is full, producer waits
            while count == BUFFER_SIZE:
                not_full.wait()

            # Insert item into buffer
            buffer[in_index] = item

            print(f"Produced: {item} at index {in_index}")

            # Move insertion index circularly
            in_index = (in_index + 1) % BUFFER_SIZE

            # Increase item count
            count += 1

            # Notify consumer that buffer has item
            not_empty.notify()

        # Simulate production delay
        time.sleep(1)


# Consumer Function
def consumer():
    global in_index, out_index, count

    while True:

        # Enter critical section
        with not_empty:

            # If buffer is empty, consumer waits
            while count == 0:
                not_empty.wait()

            # Remove item from buffer
            item = buffer[out_index]

            print(f"Consumed: {item} at index {out_index}")

            # Move removal index circularly
            out_index = (out_index + 1) % BUFFER_SIZE

            # Decrease item count
            count -= 1

            # Notify producer that buffer has space
            not_full.notify()

        # Simulate consumption delay
        time.sleep(1)


# Main Program Starts Here
if __name__ == "__main__":

    # Create producer thread
    prod_thread = threading.Thread(target=producer)

    # Create consumer thread
    cons_thread = threading.Thread(target=consumer)

    # Start both threads
    prod_thread.start()
    cons_thread.start()

    # Wait for both threads to finish
    # (they run forever here)
    prod_thread.join()
    cons_thread.join()
