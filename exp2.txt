 Exp 2 DDL Commands:


mysql> create Database Ott;


mysql> use Ott;

mysql> create table Student(roll int,name varchar(20),age int);


mysql> insert into Student values(1,"Salman",20);


mysql> select * from Student;

1 row in set (0.01 sec)

mysql> alter table Student add deptname varchar(20);

mysql> describe Student;


mysql> select * from Student;



mysql> alter table Student drop deptname;


mysql> select * from Student;


mysql> rename table Student to Stud;


mysql> select * from Stud;


mysql> truncate table Stud;


mysql> describe Stud;


mysql> select * from Stud;


mysql> drop table Stud;


mysql> drop database Ott;
