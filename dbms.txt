
Exp 6  Nested Queries in SQL:
create Database Ott;
use Ott;
create table Stud(id int,name varchar(20),mark int,div varchar(20));
insert into Stud values(1,"Rahul",85,"A");
insert into Stud values(2,"Amit",95,"A");
insert into Stud values(3,"Sneha",89,"B");
insert into Stud values(4,"Neha",99,"B");
insert into Stud values(5,"Virat",60,"A");
insert into Stud values(6,"Pawan",70,"B");

select * from Stud;



// Single row Subquery:
select name from Stud where mark = (select max(mark) from Stud);
// Multi row Subquery:
select name from Stud where mark in(select mark from Stud where mark>80);
// Correlated Subquery:
select name from Stud S1 where mark > (select min(mark) from Stud S2 where S1.div = S2.div);










Exp 7 Create Views in SQL:
create Database Ott;
use Ott;
create table Stud(id int,name varchar(20),mark int,div varchar(20));
insert into Stud values(1,"Rahul",85,"A");
insert into Stud values(2,"Amit",95,"A");
insert into Stud values(3,"Sneha",89,"B");
insert into Stud values(4,"Neha",99,"B");
insert into Stud values(5,"Virat",60,"A");
insert into Stud values(6,"Pawan",70,"B");
select * from Stud;
create view Studv as select name,id from Stud where div="A";
select * from Studv;
update Studv set name="Rakesh" where id=1;
delete from Studv where name="Virat";
drop view Studv;

Exp 9 DCL & TCL Commands:
create Database Ott;

use Ott;

create table Stud(id int,name varchar(20),mark int);

insert into Stud values(1,"Rahul",85);
insert into Stud values(2,"Amit",95);
insert into Stud values(3,"Neha",99);

select * from Stud;


// DCL Commands:


select user();

create user peter@localhost identified by "7";

show grants for peter@localhost;

grant select,update on Stud to peter@localhost;

show grants for peter@localhost;

revoke update on Stud from peter@localhost;

show grants for peter@localhost;



// TCL Commands:


(Rollback example):


start transcation;

update Stud set mark=50 where id=1;

select * from Stud;

rollback;

select * from Stud;



(Commit Example):



start transcation;

update Stud set mark=60 where id=1;

commit;

select * from Stud;




(Savepoint Example):



start transcation;

savepoint a;

update Stud set mark = 30 where id = 2;

ROLLBACK TO A;

select * from Stud;
