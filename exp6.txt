Exp 6  Nested Queries in SQL:




create Database Ott;

use Ott;

create table Stud(id int,name varchar(20),mark int,div varchar(20));

insert into Stud values(1,"Rahul",85,"A");
insert into Stud values(2,"Amit",95,"A");
insert into Stud values(3,"Sneha",89,"B");
insert into Stud values(4,"Neha",99,"B");
insert into Stud values(5,"Virat",60,"A");
insert into Stud values(6,"Pawan",70,"B");

select * from Stud;



// Single row Subquery:


select name from Stud where mark = (select max(mark) from Stud);


// Multi row Subquery:


select name from Stud where mark in(select mark from Stud where mark>80);


// Correlated Subquery:

select name from Stud S1 where mark > (select min(mark) from Stud S2 where S1.div = S2.div);
